CREATE VIEW game_switch_ps AS
SELECT 
    p.num as ps_num,
    s.num as switch_num,
    p.title,
    CASE 
        WHEN p.title IS NOT NULL AND s.title IS NOT NULL THEN 'Both'
        WHEN p.title IS NOT NULL THEN 'PS Only'
        WHEN s.title IS NOT NULL THEN 'Switch Only'
        ELSE 'None'
